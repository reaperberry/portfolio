using System.Collections;
using UnityEngine;

namespace SecondSight.Phase2
{
	/// 
	/// this script is responsible for handling the behavior of a socket that can snap objects with a specific tag into place when they enter its trigger area.
	/// It manages the snapping process, including smoothly moving the object to a designated snap point, disabling physics interactions while snapped, and applying a release force when the object is removed from the socket.
	/// The script also ensures that only one object can occupy the socket at a time and provides functionality to check if the socket is currently occupied.
	/// 
	public class Socket : MonoBehaviour
	{
		/// 
		/// tags and components to snap it to the socket. The object must have the correct tag and a Rigidbody to be snapped in place.
		/// 
		[SerializeField]
		private string _correctTag;
		/// 
		/// the transform for the object to snap to.
		/// 
		[SerializeField]
		private Transform _snapPoint;
		/// 
		/// the speed it snaps to the snap point.
		/// 
		[SerializeField]
		private float _snapSpeed = 10f;
		/// 
		/// the force that is applied when the object is released from the socket.
		/// 
		[SerializeField]
		private float _releaseForce = 5f;
		/// 
		/// this causes to give it a little spin for flair when released.
		/// 
		[SerializeField]
		private float _releaseSpin = 1f;
		/// 
		/// indicates whether the socket currently holds an object.
		/// 
		private bool _isOccupied = false;
		/// 
		/// transform of the object currently interacting / snapped to this socket.
		/// 
		private Transform _currentObject;
		/// 
		/// Represents the coroutine used to perform the snap operation.
		/// 
		private Coroutine _snapCoroutine;
		/// 
		/// this is to disable the socket collider so any other objects wont interact with it while there is an object snapped in place, and to re-enable it when the object is released.
		/// 
		private Collider _socketCollider;
		/// 
		/// this bool is to disable the collider so any other objects wont interact with it while there is an object snapped in place, and to re-enable it when the object is released.
		/// 
		private bool _socketColliderDisabled;
		/// 
		/// Contains the colliders of objects that are currently disabled.
		/// 
		private Collider[] _disabledObjectColliders;
		private const float k_SnapThreshold = 0.01f;
		private const float k_PostReleaseIgnoreDuration = 0.25f;

		private void Awake()
		{
			InitializeCollider();
		}

		private void OnTriggerEnter(Collider other)
		{
			if (!other.CompareTag(_correctTag) || _isOccupied)
				return;

			Rigidbody rb = other.attachedRigidbody;

			PrepareObjectForSnapping(rb, other);
		}

		private void OnTriggerExit(Collider other)
		{
			if (!other.CompareTag(_correctTag))
				return;

			HandleTriggerExit(other);
		}

		/// 
		/// function to check if the socket is occupied or not, for the socket checker to check if all the sockets are occupied or not.
		/// 
		/// 
		public bool IsOccupied()
		{
			return _isOccupied;
		}

		/// 
		/// Releases the currently snapped object (restores physics and re-enables colliders/grabbable).
		/// 
		public void ReleaseObject()
		{
			if (_currentObject == null)
				return;

			// Stop snapping if running to avoid conflicts with applied velocity.
			if (_snapCoroutine != null)
			{
				StopCoroutine(_snapCoroutine);
				_snapCoroutine = null;
			}

			Rigidbody rb = _currentObject.GetComponent();

			if (rb != null)
			{
				// restore gravity
				rb.useGravity = true;

				// compute a safe direction away from the snap point
				Vector3 direction =
					_currentObject.position
					- (_snapPoint != null ? _snapPoint.position : transform.position);
				if (direction.sqrMagnitude < 0.0001f)
				{
					// fallback direction if positions coincide
					direction = transform.forward;
				}
				direction.Normalize();

				// apply linear (shoot) and a small angular velocity for spin
				rb.linearVelocity = direction * _releaseForce;
				rb.angularVelocity = Random.onUnitSphere * _releaseSpin;
			}

			// Start coroutine to safely restore colliders and avoid immediate re-trigger
			StartCoroutine(ReleaseAndRestoreColliders(_currentObject));

			SetGrabbable(_currentObject, true);

			_currentObject = null;
			_isOccupied = false;
		}

		/// 
		/// function to smoothly snap the object to the snap point position and rotation, while disabling the socket collider and the object's colliders during the snap to prevent physics interactions.
		/// It also makes the object ungrabbable while snapped.
		/// The coroutine is cached so it can be stopped if the object exits the socket before fully snapping.
		/// 
		/// 
		/// 
		private IEnumerator SnapObject(Transform obj)
		{
			if (obj == null || _snapPoint == null)
				yield break;

			// rotate 90 degrees to the right relative to the snap point orientation
			Quaternion targetRotation = _snapPoint.rotation * Quaternion.Euler(0f, 90f, 0f);

			while (
				obj != null && Vector3.Distance(obj.position, _snapPoint.position) > k_SnapThreshold
			)
			{
				obj.position = Vector3.Lerp(
					obj.position,
					_snapPoint.position,
					Time.deltaTime * _snapSpeed
				);
				obj.rotation = Quaternion.Lerp(
					obj.rotation,
					targetRotation,
					Time.deltaTime * _snapSpeed
				);
				yield return null;
			}

			if (obj != null)
			{
				obj.position = _snapPoint.position;
				obj.rotation = targetRotation;

				// disable socket collider and the object's colliders now that it's fully snapped
				DisableSocketCollider();
				DisableObjectColliders(obj);
			}

			// clear coroutine handle
			_snapCoroutine = null;
		}

		/// 
		/// Disable/enable common grab components by name on the object and its children.
		/// 
		/// 
		/// 
		private void SetGrabbable(Transform obj, bool grabbable)
		{
			if (obj == null)
				return;

			Component[] comps = obj.GetComponentsInChildren(true);

			foreach (var comp in comps)
			{
				if (comp == null)
					continue;

				string tname = comp.GetType().Name;

				if (
					tname == "XRGrabInteractable"
					|| tname == "OVRGrabbable"
					|| tname == "Grabbable"
					|| tname == "GrabInteractable"
				)
				{
					if (comp is Behaviour b)
					{
						b.enabled = grabbable;
					}
				}
			}
		}

		/// 
		/// Disable the socket's collider so it no longer triggers while object is snapped.
		/// 
		private void DisableSocketCollider()
		{
			if (_socketCollider != null && _socketCollider.enabled)
			{
				_socketCollider.enabled = false;
				_socketColliderDisabled = true;
			}
		}

		/// 
		/// gets the collider component of the socket and initializes the state for managing collider interactions during snapping and releasing.
		/// This is called in Awake to set up the initial state of the socket's collider management.
		/// 
		private void InitializeCollider()
		{
			_socketCollider = GetComponent();
			_socketColliderDisabled = false;
			_disabledObjectColliders = null;
		}

		/// 
		/// sets the velocity of the object to zero and disables gravity to prepare it for snapping, and also sets the current object reference and occupied state.
		/// This is called when an object enters the socket trigger and is valid for snapping.
		/// 
		/// 
		/// 
		private void PrepareObjectForSnapping(Rigidbody rb, Collider other)
		{
			if (rb != null)
			{
				// zero motion using project-specific API
				rb.linearVelocity = Vector3.zero;
				rb.angularVelocity = Vector3.zero;
				rb.useGravity = false;
			}

			_currentObject = other.transform;

			// Make the object ungrabbable immediately
			SetGrabbable(other.transform, false);

			_isOccupied = true;
			_snapCoroutine = StartCoroutine(SnapObject(other.transform));
		}

		/// 
		/// checks the state of the socket when an object exits the trigger. If the exiting object is the one currently snapped, it releases it and clears the socket state.
		/// If it's a different object (e.g., passing through), it ensures that object remains grabbable and doesn't affect the socket's occupied state.
		/// 
		/// 
		private void HandleTriggerExit(Collider other)
		{
			// Stop any ongoing snap so it won't resume after exit
			if (_snapCoroutine != null)
			{
				StopCoroutine(_snapCoroutine);
				_snapCoroutine = null;
			}

			// If the object leaving is the one we were handling, clear socket state.
			if (_currentObject == other.transform)
			{
				SetGrabbable(other.transform, true);
				_isOccupied = false;
				_currentObject = null;
			}
			else
			{
				// Ensure other passing objects remain grabbable
				SetGrabbable(other.transform, true);
			}
		}
	}
}
